\begin{frame}[parent={ie:agenda}, hasnext=true, hasprev=true]
	\frametitle{Modelo de ciclo de vida}

	\begin{block:concept}{Definição}
		\begin{itemize}
			\item Estabelecer as etapas de desenvolvimento de software.
			\begin{itemize}
				\item Cada etapa agrega um ou mais processos.
			\end{itemize}
			
			\item Estabelecer a ordem em que essas etapas são executadas.
			\begin{itemize}
				\item Critérios para mudar de etapa.
			\end{itemize}
		\end{itemize}
	\end{block:concept}
	
	\note{
		\begin{itemize}
			\item Embora defina as etapas, o modelo não define \textbf{como} será
			feito dentro de cada etapa. Isto é papel de métodos de desenvolvimento
			de software.
			
			\item Ao estabelecer etapas e critérios para alternar entre as etapas,
			é estabelecida uma estrutura para o gerenciamento de projeto.
		\end{itemize}
	}
\end{frame}
